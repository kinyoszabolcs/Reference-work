125  public void MoveEnemies()
126  {

129  this.AddEnemyWave();

131  }
132 
133  try
134  {
135  foreach (Enemy item in this.Model.Enemies)
136  {
137  item.ChangeY(item.Dy);
138  if (item.Area.Bottom > Config.Height)
139  {
140  this.Model.Enemies.Remove(item);
141  this.Gameover = true;
142  }
143 
144  foreach (var bullet in this.Model.Bullets)
145  {

147  {
